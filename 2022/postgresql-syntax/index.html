postgresql语法

作者： qingling0910 时间： June 26, 2022 分类：

1、登录
sudo -I -u postgres
2、进入postgresql
psql
3、创建数据库
create database database_name;
4、创建表
1)先进入数据库
\l data_name
2)创建表

create table table_name(column_name1 data_type,column_name2 data_type,….);	     
   postgres=# create table company4(id int primary key not null,name text not null,age int not null,address char(50),salary real);

5、查看表格信息
\d table_name;

10、删除特定模式下的表
drop table schema_name.table_name;

postgres=# drop table qingling.company;

11、删除schema下的所有对象
drop schema schema_name cascade;

postgres=# drop schema qingling cascade;
NOTICE:  drop cascades to table qingling.company

12、删除schema
drop schema schema_name;

postgres=# drop schema qingling;

13、插入数据

insert into table_name（column_name1，column_name2，。。。。）values(value1,value2,...);

14、with子句用法

WITH
   name_for_summary_data AS (
      SELECT Statement)
   SELECT columns
   FROM name_for_summary_data
   WHERE conditions <=> (
      SELECT column
      FROM name_for_summary_data)
   [ORDER BY columns]

postgres=# WITH RECURSIVE t(n) AS (VALUES (0) UNION ALL SELECT SALARY FROM COMPANY WHERE SALARY < 20000) SELECT sum(n) FROM t;
  sum  
-------
 40000
(1 row)

15、删除约束
1）先找到约束名称
\d table_nam;

postgres=# \d company5
                 Table "public.company5"
 Column  |     Type      | Collation | Nullable | Default 
---------+---------------+-----------+----------+---------
 id      | integer       |           | not null | 
 name    | text          |           | not null | 
 age     | integer       |           | not null | 
 address | character(50) |           |          | 
 salary  | real          |           |          | 
Indexes:
    "company5_pkey" PRIMARY KEY, btree (id)
Check constraints:
    "company5_salary_check" CHECK (salary > 0::double precision)

2)删除约束
alter table table_name drop constraint 约束名称;

postgres=# alter table company5 drop constraint if exists company5_salary_check;

16、创建索引
CREATE INDEX index_name ON table_name(column_name); 

postgres=# create index qingling_index on company(name);

17、组合索引

CREATE INDEX index_name ON table_name (column1_name, column2_name);

18、唯一索引

CREATE UNIQUE INDEX index_name on table_name (column_name); 

postgres=# create unique index unique_index on department(emp_id);

19、局部索引

CREATE INDEX index_name on table_name (conditional_expression);

21、索引删除索引

drop index index_name; 

postgres=# drop index qingling_index;

22、用 ALTER TABLE 在一张已存在的表上添加列的语法如下：

ALTER TABLE table_name ADD column_name datatype;

postgres=# alter table department add column address char(50);

23、在一张已存在的表上 DROP COLUMN（删除列），语法如下：

alter table table_name drop column column_name; 

postgres=# alter table department drop address;

24、修改表中某列的 DATA TYPE（数据类型），语法如下(char转换给int)(int转换给varcahr)：

ALTER TABLE table_name ALTER COLUMN column_name TYPE datatype;

postgres=# alter table department alter column address type char(50);

25、修改表中某列的 DATA TYPE（数据类型），语法如下(char转换给int)：

alter table table_name alter column column_name type int using column_name::int; 

postgres=# alter table department alter column address type int using address::int;

26、给表中某列添加 NOT NULL 约束，语法如下：

alter table table_name ALTER column_name set NOT NULL; 

postgres=# alter table department alter column emp_id  set not null;

27、删除表

TRUNCATE TABLE  table_name;

postgres=# truncate table company5;

28、创建视图

create view view_name as 条件; 

postgres=# create view company_view as select id,name,age from company;

29、删除视图

drop view view_name; 

postgres=# drop view company_view;

30、授权给用户

GRANT privilege [, ...] ON object [, ...] TO { PUBLIC | GROUP group | username } 

postgres=# grant all on company to runoob;

31、撤销权限

postgres=# revoke all on company from runoob;
REVOKE
